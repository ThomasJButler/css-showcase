@media (max-width: 768px) {
    .table-responsive thead {
        display: none;
    }
    
    .table-responsive tr {
        display: block;
        margin-bottom: 1rem;

        border-radius: 0.5rem;

        display: block;
        text-align: right;
        padding-left: 50%;
        position: relative;
    }

        position: absolute;
        left: 1rem;
        font-weight: 600;
    }
}
